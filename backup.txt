const userLogin = async(req, res)=>{
    try {
        
        let {email, password} = req.body;

        let userLoginAccount = await LoginModel.findOne({email});

        console.log("userLoginAccount ", userLoginAccount)

        if(!!userLoginAccount && userLoginAccount.isAccountLocked){
            return res.status(403).json({
                message : "Opps,Account locked for 15 mins. Try again later.",
                success : false
            });
        }

        let user = await UserModel.findOne({email, isEmailVerified: true});

        if(!user){
            return res.status(404).json({
                message : "Invalid User or User not verified",
                success: false
            });
        }

        // checking for password
        let isPasswordCorrect = await user.isPasswordCorrect(password);

        if(!isPasswordCorrect && userLoginAccount.failedAttempts <=4){

            userLoginAccount.failedAttempts+=1;
            userLoginAccount.lastAttemptAt = Date.now();

            await userLoginAccount.save();

            return res.status(400).json({
                message : "Invalid Credentials",
                success : false
            })
        }else {
            userLoginAccount.isAccountLocked = true;
            userLoginAccount.lastAttemptAt = Date.now();
            userLoginAccount.lockedUntil = Date.now() +(15*60*1000); // 15 mins
            await userLoginAccount.save();
            return res.status(403).json({
                message : "Your Account has been locked for 15 mins. Try after 15 mins",
                success : false
            })
        } 

        let payload = {
            id : user._id,
            email : user.email
        };

        let accessToken = jwt.sign(payload,process.env.JWT_SECRET_KEY, {expiresIn:"2h"});
        let refreshToken = jwt.sign(payload,process.env.JWT_SECRET_KEY, {expiresIn:"7d"});

        user.refreshToken = refreshToken;
        user.status = "online";
        await user.save();

        res.status(200).cookie("token", accessToken,{ 
            // httpOnly: true,
            secure: true,          
            sameSite: 'strict',
            maxAge: 2 * 60 * 60 * 1000
        }).status(200).json({
            message : `Login Success. Welcome ${user.name}`,
            success : true,
        });

    } catch (error) {
        res.status(500).json({ 
            message: "Server Error", 
            error: error.message, 
            success: false 
        });
    }
}
